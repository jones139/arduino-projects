019public class BluetoothArduino extends Thread {

036        return __blue == null ? new BluetoothArduino(n) : __blue;

044    private  BluetoothArduino(String Name){
045        __blue = this;
046        try {
047            for(int i = 0; i < 2048; i++){
048                mMessages.add("");
049            }
050            robotName = Name;
051            mBlueAdapter = BluetoothAdapter.getDefaultAdapter();
052            if (mBlueAdapter == null) {
053                LogError("\t\t[#]Phone does not support bluetooth!!");
054                return;
055            }
056            if (!isBluetoothEnabled()) {
057              LogError("[#]Bluetooth is not activated!!");
058            }
059
060            Set<BluetoothDevice> paired = mBlueAdapter.getBondedDevices();
061            if (paired.size() > 0) {
062                for (BluetoothDevice d : paired) {
063                    if (d.getName().equals(robotName)) {
064                        mBlueRobo = d;
065                        robotFound = true;
066                        break;
067                    }
068                }
069            }
070            
071            if (!robotFound)
072                LogError("\t\t[#]There is not robot paired!!");
073
074        }catch (Exception e){
075            LogError("\t\t[#]Erro creating Bluetooth! : " + e.getMessage());
076        }
077
078    }
079
080    BluetoothArduino(){
081        this("Arduino-Robot");
082    }
083    
084    public boolean isBluetoothEnabled(){
085      return mBlueAdapter.isEnabled();
086    }
087
088    public boolean Connect(){
089        if(!robotFound)
090            return false;
091        try{
092            LogMessage("\t\tConncting to the robot...");
093
094            UUID uuid = UUID.fromString("00001101-0000-1000-8000-00805f9b34fb");
095            mBlueSocket = mBlueRobo.createRfcommSocketToServiceRecord(uuid);
096            mBlueSocket.connect();
097            mOut = mBlueSocket.getOutputStream();
098            mIn = mBlueSocket.getInputStream();
099            connected = true;
100            this.start();
101            LogMessage("\t\t\t" + mBlueAdapter.getName());
102            LogMessage("\t\tOk!!");
103            return true;
104
105        }catch (Exception e){
106            LogError("\t\t[#]Error while conecting: " + e.getMessage());
107            return false;
108        }
109    }
110
111    public void run(){
112
113        while (true) {
114            if(connected) {
115                try {
116                    byte ch, buffer[] = new byte[1024];
117                    int i = 0;
118
119                    String s = "";
120                    while((ch=(byte)mIn.read()) != DELIMITER){
121                        buffer[i++] = ch;
122                    }
123                    buffer[i] = '\0';
124
125                    final String msg = new String(buffer);
126
127                    MessageReceived(msg.trim());
128                    LogMessage("[Blue]:" + msg);
129
130                } catch (IOException e) {
131                    LogError("->[#]Failed to receive message: " + e.getMessage());
132                }
133            }
134        }
135    }
136
137    private void MessageReceived(String msg){
138        try {
139
140            mMessages.add(msg);
141            try {
142                this.notify();
143            }catch (IllegalMonitorStateException e){
144                //
145            }
146        } catch (Exception e){
147            LogError("->[#] Failed to receive message: " + e.getMessage());
148        }
149    }

151    public boolean hasMensagem(int i){
152        try{
153            String s = mMessages.get(i);
154            if(s.length() > 0)
155                return true;
156            else
157                return false;
158        } catch (Exception e){
159            return false;
160        }
161    }

175   public String getLastMessage(){
176     if(countMessages() == 0)

181    public void SendMessage(String msg){
182        try {
183            if(connected) {
184                mOut.write(msg.getBytes());
185            }
186
187        } catch (IOException e){
188            LogError("->[#]Error while sending message: " + e.getMessage());
189        }
190    }
