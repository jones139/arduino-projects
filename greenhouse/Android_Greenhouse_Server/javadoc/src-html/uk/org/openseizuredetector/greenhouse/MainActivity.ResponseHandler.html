134    @Override
135    public boolean onOptionsItemSelected(MenuItem item) {
136        Log.v(TAG,"Option "+item.getItemId()+" selected");
137        Log.v(TAG,"action_start_stop = "+R.id.action_start_stop);
138        switch (item.getItemId()) {
139        case R.id.action_start_stop:
140            // Respond to the start/stop server menu item.
141            Log.v(TAG,"action_sart_stop");
142            if (mBound) {
143                Log.v(TAG,"Stopping Server");
144                unbindFromServer();
145                stopServer();
146            } else {
147                Log.v(TAG,"Starting Server");
148                startServer();
149                // and bind to it so we can see its data
150                bindToServer();
151            }
152            return true;
153        case R.id.action_settings:
154            Log.v(TAG,"action_settings");
155            try {
156                Intent prefsIntent = new Intent(
157                                                MainActivity.this,
158                                                PrefActivity.class);
159                this.startActivity(prefsIntent);
160            } catch (Exception ex) {
161                Log.v(TAG,"exception starting settings activity "+ex.toString());
162            }
163            return true;
164        default:
165            return super.onOptionsItemSelected(item);
166        }
167    }

169    @Override
170    protected void onStart() {
171        super.onStart();
172        SharedPreferences SP = PreferenceManager
173            .getDefaultSharedPreferences(getBaseContext());
174        boolean audibleAlarm = SP.getBoolean("AudibleAlarm",true);
175        Log.v(TAG,"onStart - auidbleAlarm = "+audibleAlarm);
176
177        TextView tv;
178        tv = (TextView) findViewById(R.id.versionTv);
179        String versionName = "unknown";
180        // From http://stackoverflow.com/questions/4471025/
181        //         how-can-you-get-the-manifest-version-number-
182        //         from-the-apps-layout-xml-variable
183        final PackageManager packageManager = getPackageManager();
184        if (packageManager != null) {
185            try {
186                PackageInfo packageInfo = packageManager.getPackageInfo(getPackageName(), 0);
187                versionName = packageInfo.versionName;
188            } catch (PackageManager.NameNotFoundException e) {
189                Log.v(TAG,"failed to find versionName");
190                versionName = null;
191            }
192        }
193        tv.setText("GreenHouseServer Version "+versionName);
194
195        if (!isServerRunning()) {
196            Log.v(TAG,"Server not Running - Starting Server");
197            startServer();
198        } else {
199            Log.v(TAG,"Server Already Running OK");
200        }
201        // and bind to it so we can see its data
202        bindToServer();
203
204    }

212  /** Defines callbacks for service binding, passed to bindService() */
213    private ServiceConnection mConnection = new ServiceConnection() {
214
215        @Override
216        public void onServiceConnected(ComponentName className,
217                IBinder service) {
218            // We've bound to LocalService, cast the IBinder and get LocalService instance
219            GreenHouseServer.GreenHouseBinder binder = (GreenHouseServer.GreenHouseBinder) service;
220            mGreenHouseServer = binder.getService();
221            mBound = true;
222            if (mGreenHouseServer!=null) {
223                Log.v(TAG,"onServiceConnected() - Asking server to update its settings");
224                mGreenHouseServer.updatePrefs();
225            }
226            else {
227                Log.v(TAG,"onServiceConnected() - mGreenHouseServer is null - this is wrong!");
228            }
229        }
230
231        @Override
232        public void onServiceDisconnected(ComponentName arg0) {
233            Log.v(TAG,"onServiceDisonnected()");
234            mBound = false;
235        }
236    };

239    /**
240     * bind to an already running server.
241     */
242    private void bindToServer() {
243        Log.v(TAG,"bindToServer() - binding to SdServer");
244        Intent intent = new Intent(this,GreenHouseServer.class);
245        bindService(intent,mConnection, Context.BIND_AUTO_CREATE);
246    }
247
248    /** 
249     * unbind from server
250     */
251    private void unbindFromServer() {
252        // unbind this activity from the service if it is bound.
253        if (mBound) {
254            Log.v(TAG,"unbindFromServer() - unbinding");
255            try {
256                unbindService(mConnection);
257                mBound = false;
258            } catch (Exception ex) {
259                Log.e(TAG,"unbindFromServer() - error unbinding service - "+ex.toString());
260            }
261        } else {
262            Log.v(TAG,"unbindFromServer() - not bound to server - ignoring");
263        }
264    }

266    /**
267     * Start the GreenHouseServer service
268     */
269    private void startServer() {
270        // Start the server
271        greenHouseServerIntent = new Intent(MainActivity.this,GreenHouseServer.class);
272        greenHouseServerIntent.setData(Uri.parse("Start"));
273        getApplicationContext().startService(greenHouseServerIntent);
274
275        // Change the action bar icon to show the option to stop the service.
276        if (mOptionsMenu!=null) {
277            Log.v(TAG,"Changing menu icons");
278            MenuItem menuItem = mOptionsMenu.findItem(R.id.action_start_stop);
279            menuItem.setIcon(R.drawable.stop_server);
280            menuItem.setTitle("Stop Server");
281        } else {
282            Log.v(TAG,"mOptionsMenu is null - not changing icons!");
283        }
284    }
285
286    /**
287     * Stop the GreenHouseServer service
288     */
289    private void stopServer() {
290        Log.v(TAG,"stopping Server...");
291
292        // then send an Intent to stop the service.
293        greenHouseServerIntent = new Intent(MainActivity.this,
294                                            GreenHouseServer.class);
295        greenHouseServerIntent.setData(Uri.parse("Stop"));
296        getApplicationContext().stopService(greenHouseServerIntent);
297
298        // Change the action bar icon to show the option to start the service.
299        if (mOptionsMenu!=null) {
300            Log.v(TAG,"Changing action bar icons");
301            mOptionsMenu.findItem(R.id.action_start_stop).setIcon(R.drawable.start_server);
302            mOptionsMenu.findItem(R.id.action_start_stop).setTitle("Start Server");
303        } else {
304            Log.v(TAG,"mOptionsMenu is null, not changing icons!");
305        }
306
307    }

309    /**
310     * Based on http://stackoverflow.com/questions/7440473/android-how-to-check-if-the-intent-service-is-still-running-or-has-stopped-running
311     */
312    public boolean isServerRunning() {
313        //Log.v(TAG,"isServerRunning()................");
314        ActivityManager manager = 
315            (ActivityManager) getSystemService(ACTIVITY_SERVICE);
316        for (RunningServiceInfo service : 
317                 manager.getRunningServices(Integer.MAX_VALUE)) {
318            //Log.v(TAG,"Service: "+service.service.getClassName());
319            if ("uk.org.openseizuredetector.greenhouse.GreenHouseServer"
320                .equals(service.service.getClassName())) {
321                //Log.v(TAG,"Yes!");
322            return true;
323            }
324        }
325        //Log.v(TAG,"No!");
326        return false;
327    }

330    /** get the ip address of the phone.
331     * Based on http://stackoverflow.com/questions/11015912/how-do-i-get-ip-address-in-ipv4-format
332     */
333    public String getLocalIpAddress() {
334        try {
335            for (Enumeration<NetworkInterface> en = NetworkInterface
336                     .getNetworkInterfaces(); en.hasMoreElements();) {
337                NetworkInterface intf = en.nextElement();
338                for (Enumeration<InetAddress> enumIpAddr = intf
339                         .getInetAddresses(); enumIpAddr.hasMoreElements();) {
340                    InetAddress inetAddress = enumIpAddr.nextElement();
341                    //Log.v(TAG,"ip1--:" + inetAddress);
342                    //Log.v(TAG,"ip2--:" + inetAddress.getHostAddress());
343                
344                    // for getting IPV4 format
345                    if (!inetAddress.isLoopbackAddress() 
346                        && InetAddressUtils.isIPv4Address(
347                                 inetAddress.getHostAddress())) {
348                    
349                        String ip = inetAddress.getHostAddress().toString();
350                        //Log.v(TAG,"ip---::" + ip);
351                        return ip;
352                    }
353                }
354            }
355        } catch (Exception ex) {
356            Log.e("IP Address", ex.toString());
357        }
358        return null;
359    }

363    /*
364     * updateServerStatus - called by the uiTimer timer periodically.
365     * requests the ui to be updated by calling serverStatusRunnable.
366     */
367    private void updateServerStatus() {
368        serverStatusHandler.post(serverStatusRunnable);
369    }
370        
371    /*
372     * serverStatusRunnable - called by updateServerStatus - updates the
373     * user interface to reflect the current status received from the server.
374     */
375    final Runnable serverStatusRunnable = new Runnable() {
376            public void run() {
377                TextView tv;
378                tv = (TextView) findViewById(R.id.textView1);
379                if (isServerRunning()) {
380                    tv.setText("Server Running OK");
381                    tv.setBackgroundColor(okColour);
382                    tv = (TextView)findViewById(R.id.textView2);
383                    tv.setText("Access Server at http://"
384                                   +getLocalIpAddress()
385                                   +":8080");
386                    tv.setBackgroundColor(okColour);
387                } else {
388                    tv.setText("*** Server Stopped ***");
389                    tv.setBackgroundColor(alarmColour);
390                }
391
392
393                try {
394                    if (mBound) {
395                        tv = (TextView) findViewById(R.id.alarmTv);
396                        if (mGreenHouseServer.mGd.alarmState==0) {
397                            tv.setText(mGreenHouseServer.mGd.alarmPhrase);
398                            tv.setBackgroundColor(okColour);
399                        }
400                    }
401                } catch (Exception e) {
402                    Log.v(TAG,"ServerStatusRunnable: Exception - "+e.toString());
403                }
404            }
405        };
